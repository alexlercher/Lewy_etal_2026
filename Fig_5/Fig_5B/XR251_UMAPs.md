data$manual <- dplyr::case_when(
  data$manual == "VSMCs" ~ "Fibroblast",
  data$manual == "Fibroblast" ~ "Fibroblast-like",
  data$manual == "Ventricular Ephithelial cells" ~ "Ventricular Epithelial cells",
  TRUE ~ as.character(data$manual)              # keep everything else
) %>%
  factor() %>%                                  # convert back to factor
  droplevels()

# rename manual to cell_type (maria called it manual because we annotated manually)
data@meta.data <- data@meta.data %>%
  dplyr::rename(cell_type = manual)

# reset Idents()
Idents(data) <- data$cell_type

#save my modified Seurat object
saveRDS(data, file = "input/seu_obj_v12192025_AL.rds")

## INSPECT AND SAVE UMAPs GENERATED BY MARIA

``` r
# define colors
cols <- c(
  "#E69F00","#56B4E9","#009E73","#F0E000",
  "#0072B2","#D55E00","#CC79A7","#999999",
  "#1B9E77","#D90F02","#7570B3","#E7298A"
)

# UMAP by cell type
p1 <- DimPlot(data,
              reduction = "umap",
              group.by = "cell_type",
              label = TRUE,
              repel = TRUE,
              cols = cols) +
  theme(aspect.ratio = 1) +
  labs(title = "cell types")

print(p1)
```
