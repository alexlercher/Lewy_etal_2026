## CALCULATE AND VISUALIZE CELLS PER CLUSTER PER SAMPLE

``` r
# summarize cell counts and calculate proportion per cluster per sample
cell_counts <- data@meta.data %>%
  count(cell_type, condition) %>%
  tidyr::pivot_wider(
    names_from = condition,
    values_from = n,
    values_fill = 0
  ) %>%
  mutate(total_cells = rowSums(across(where(is.numeric)))) %>%
  mutate(across(
    where(is.numeric),
    ~ .x / total_cells,
    .names = "prop_{.col}"
  ))

cell_counts %>%
  kable(
    format = "markdown",
    digits = 2,
    caption = "Cells per cluster"
  )
```
